305 template<typename T>
306 T convertToSeconds(const std::string &unit, T value) {
307  T seconds;
308  if (unit == "min") {
309  seconds = value * 60;
310  } else if (unit == "h") {
311  std::string new_unit = "min";
312  seconds = convertToSeconds(new_unit, value * 60);
313  } else if (unit == "s" || unit == "sec") {
314  seconds = value;
315  } else if (isScalable(unit, "s")) {
316  double scaled = value * getSIScaling(unit, "s");
317  seconds = static_cast<T>(std::is_integral<T>::value ? std::round(scaled) : scaled);
318  } else {
319  std::cerr << "[nix::util::convertToSeconds] Warning: given unit is not supported!" << std::endl;
320  seconds = value;
321  }
322  return seconds;
323 }

333 template<typename T>
334 T convertToKelvin(const std::string &unit, T value) {
335 
336  if (unit == "°K" || unit == "K") {
337  return value;
338  }
339 
340  double temperature;
341  if (unit == "°C" || unit == "C") {
342  temperature = value + 273.15;
343  } else if (unit == "°F" || unit == "F") {
344  temperature = (value - 32) * 5.0/9 + 273.15;
345  } else if (isScalable(unit, "K")) {
346  temperature = value * getSIScaling(unit, "K");
347  } else {
348  std::cerr << "[nix::util::convertToKelvin] Warning: given unit is not supported" << std::endl;
349  return value;
350  }
351 
352  return static_cast<T>(std::is_integral<T>::value ? std::round(temperature) : temperature);
353 }
