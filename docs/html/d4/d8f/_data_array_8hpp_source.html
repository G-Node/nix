317  const std::string &unit="") {
318  if (ticks.size() == 0) {

320  "DataArray::appendRangeDimension");
321  }
322  RangeDimension dim = backend()->createRangeDimension(backend()->dimensionCount() + 1, ticks);

337  RangeDimension appendAliasRangeDimension() {
338  if (this->dataExtent().size() > 1) {
339  throw nix::InvalidDimension("AliasRangeDimensions only allowed for 1D numeric DataArrays!",
340  "DataArray::appendAliasRangeDimension");
341  }
342  if (!nix::data_type_is_numeric(this->dataType())) {
343  throw nix::InvalidDimension("AliasRangeDimensions are only allowed for 1D numeric DataArrays!",
344  "DataArray::appendAliasRangeDimension");
345  }
346  if (dimensionCount() > 0) {
347  throw nix::InvalidDimension("Cannot append additional alias dimension. There must only be one!",
348  "DataArray::appendAliasRangeDimension");
349  }
350  if (this->unit()) {
351  std::string unit = *this->unit();
352  if(!(util::isSIUnit(unit) || util::isCompoundSIUnit(unit))) {
353  throw nix:: InvalidUnit("AliasRangeDimensions are only allowed when SI or composites of SI units are used.",
354  "DataArray::appendAliasRangeDimension");
355  }
356  }
357  return backend()->createAliasRangeDimension();
358  }
