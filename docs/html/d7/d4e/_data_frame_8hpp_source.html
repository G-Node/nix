251  template<typename T>
252  void readColumn(const std::string &name,
253  std::vector<T> &vals,
254  bool resize = false,
255  ndsize_t offset = 0) {
256  Hydra<std::vector<T>> hydra(vals);
257 
258  size_t count = vals.size();
259  if (resize) {
260  ndsize_t n_rows = rows();
261  if (offset > n_rows) {
262  throw OutOfBounds("offset > number of rows");
263  }
264 
265  // >= 0, as per the check above
266  n_rows -= offset;
267 
268  count = check::fits_in_size_t(n_rows, "n > sizeof(size_t)");
269  }
270 
271  // OOB check done by called readColumn below
272  return this->readColumn<T>(name, vals, count, resize, offset);
273  }
