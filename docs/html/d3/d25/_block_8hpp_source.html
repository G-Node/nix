379  template<typename T>
380  DataArray createDataArray(const std::string &name,
381  const std::string &type,
382  const T &data,
383  DataType data_type=DataType::Nothing,
384  const Compression &compression=Compression::Auto) {
385  const Hydra<const T> hydra(data);
386 
387  if (data_type == DataType::Nothing) {
388  data_type = hydra.element_data_type();
389  }
390 
391  const NDSize shape = hydra.shape();
392  DataArray da = createDataArray(name, type, data_type, shape, compression);
393 
394  const NDSize offset(shape.size(), 0);
395  da.setData(data, offset);
396 
397  return da;
